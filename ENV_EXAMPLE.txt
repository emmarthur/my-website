# Environment Variables Example
# Copy this to .env.local and fill in your values

# Unsplash API Key (Optional)
# Get your free API key from https://unsplash.com/developers
# 
# IMPORTANT: Use the "Access Key" (Application ID), NOT the Secret Key!
# The Access Key is safe for client-side use.
# The Secret Key should NEVER be exposed in client-side code.
#
NEXT_PUBLIC_UNSPLASH_ACCESS_KEY=your_access_key_here

# Replicate API Token (For AI Image Generation)
# Get your API token from https://replicate.com/account/api-tokens
# This enables Midjourney-style image generation
#
REPLICATE_API_TOKEN=your_replicate_api_token_here

# Note: The website works without these API keys
# The animated particle background and gradients will still function beautifully
# AI image generation requires the Replicate API token
